\begin{tikzpicture}
\tikzset{normalline/.style={thin}}

\pgfmathsetmacro{\xlo}{-1.};
\pgfmathsetmacro{\xhi}{5.};
\pgfmathsetmacro{\ylo}{-1.};
\pgfmathsetmacro{\yhi}{3.};
\pgfmathsetmacro{\rad}{.07pt};
\pgfmathsetmacro{\margin}{.5pt};
\pgfmathsetmacro{\linescale}{20};

\clip ({\xlo-\margin},{\ylo-\margin}) rectangle ({\xhi+\margin},{\yhi+\margin});

\path [name path=left-egde] (\xlo,\ylo) -- (\xlo,\yhi);
\path [name path=right-egde] (\xhi,\ylo) -- (\xhi,\yhi);
\path [name path=bottom-egde] (\xlo,\ylo) -- (\xhi,\ylo);
\path [name path=top-egde] (\xlo,\yhi) -- (\xhi,\yhi);

\coordinate [label={[xshift=5.,yshift=1.]:$A$}] (A) at (0,2);
\coordinate [label={[xshift=2.,yshift=1.]:$B$}] (B) at (3,2);
\coordinate [label={[xshift=-1.,yshift=-19.]:$C$}] (C) at (1,0);
\coordinate [label={[xshift=-4.,yshift=-19.]:$D$}] (D) at (3.7,.4);

\path [name path=line_A-B] ($ (A)!\linescale!(B) $) -- ($ (B)!\linescale!(A) $);
\path [name path=line_A-C] ($ (A)!\linescale!(C) $) -- ($ (C)!\linescale!(A) $);
\path [name path=line_B-C] ($ (B)!\linescale!(C) $) -- ($ (C)!\linescale!(B) $);
\path [name path=line_B-D] ($ (B)!\linescale!(D) $) -- ($ (D)!\linescale!(B) $);
\path [name path=line_C-D] ($ (C)!\linescale!(D) $) -- ($ (D)!\linescale!(C) $);

\path [name intersections={of=line_A-B and left-egde, by=left_l}];
\path [name intersections={of=line_A-B and right-egde, by=right_l}];
\path [name intersections={of=line_A-C and bottom-egde, by=left_n}];
\path [name intersections={of=line_A-C and top-egde, by=right_n}];
\path [name intersections={of=line_B-C and bottom-egde, by=left_m}];
\path [name intersections={of=line_B-C and top-egde, by=right_m}];
\path [name intersections={of=line_B-D and top-egde, by=left_k}];
\path [name intersections={of=line_B-D and bottom-egde, by=right_k}];
\path [name intersections={of=line_C-D and left-egde, by=left_h}];
\path [name intersections={of=line_C-D and right-egde, by=right_h}];

\draw[normalline] (left_l) -- (right_l) node [right] {$l$};
\draw[normalline] (left_m) -- (right_m) node [above] {$m$};
\draw[normalline] (left_h) -- (right_h) node [right] {$h$};
\draw[normalline] (left_n) -- (right_n) node [above] {$n$};
\draw[normalline] (left_k) -- (right_k) node [right] {$n$};

\fill (A) circle (\rad);
\fill (B) circle (\rad);
\fill (C) circle (\rad);
\fill (D) circle (\rad);
\end{tikzpicture}
